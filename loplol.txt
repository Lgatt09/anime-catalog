<script>
const animeList = document.getElementById("animeList");
const loading = document.getElementById("loading");
const searchInput = document.getElementById("search");
let allAnimes = [];

// Crear tarjeta anime
function createAnimeCard(anime) {
  const card = document.createElement("div");
  card.classList.add("anime-card");
  card.setAttribute('tabindex', '0');

  card.innerHTML = `
    <img src="${anime.images.jpg.image_url}" alt="Portada de ${anime.title}" loading="lazy" />
    <div class="anime-content">
      <div class="anime-name" title="${anime.title}">${anime.title}</div>
      <div class="links">
        <a href="https://jkanime.net/?s=${encodeURIComponent(anime.title)}" target="_blank" rel="noopener noreferrer" aria-label="Buscar ${anime.title} en Jkanime">Jkanime</a>
        <a href="https://tioanime.com/?s=${encodeURIComponent(anime.title)}" target="_blank" rel="noopener noreferrer" aria-label="Buscar ${anime.title} en TioAnime">TioAnime</a>
      </div>
    </div>
  `;
  return card;
}

// Mostrar lista filtrada
function showAnimes(filter = "") {
  animeList.innerHTML = "";
  const filtered = allAnimes.filter(anime => anime.title.toLowerCase().includes(filter.toLowerCase()));
  if(filtered.length === 0) {
    animeList.innerHTML = "<p class='no-results'>No se encontraron animes.</p>";
  } else {
    filtered.forEach(anime => {
      animeList.appendChild(createAnimeCard(anime));
    });
  }
}

// Cargar animes top usando Jikan API (MyAnimeList unofficial)
async function loadTopAnimes() {
  try {
    loading.style.display = "block";
    animeList.innerHTML = "";

    const pagesToLoad = 4;
    let fetchedAnimes = [];

    for(let page = 1; page <= pagesToLoad; page++) {
      const response = await fetch(`https://api.jikan.moe/v4/top/anime?page=${page}&limit=25`);
      const data = await response.json();
      fetchedAnimes = fetchedAnimes.concat(data.data);
    }

    allAnimes = fetchedAnimes;
    showAnimes();

  } catch (error) {
    animeList.innerHTML = "<p class='no-results'>Error al cargar animes. Intenta recargar la página.</p>";
  } finally {
    loading.style.display = "none";
  }
}

// Evento búsqueda
searchInput.addEventListener("input", e => {
  showAnimes(e.target.value);
});

// Inicio
loadTopAnimes();
</script>
